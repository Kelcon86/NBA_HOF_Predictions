-- Creating tables for nba-hof-project DB

CREATE TABLE public.career_duration
(
    index integer NOT NULL,
    name_first_last character varying(40),
    name_last_first character varying(40),
    from_year integer,
    to_year integer,
    PRIMARY KEY (index)
);


CREATE TABLE public.draft_postion
(
    index integer NOT NULL,
    player character varying(40),
    year integer,
   	pick integer,
    PRIMARY KEY (index)
);

CREATE TABLE public.rookies_df
(
    index integer NOT NULL,
   	name character varying(40),
	HallofFameClass character varying(20),
    year_drafted integer,
   	GP integer,
	MIN numeric,
	PTS numeric,
	FGM numeric,
	FGA numeric,
	"FG%" numeric,
	"3PMade" numeric,
	"3PA" numeric,
	"3P%" numeric,
	FTM numeric,
	FTA numeric,
	"FT%" numeric,
	OREB numeric,
	DREB numeric,
	REB numeric,
	AST numeric,
	STL numeric,
	BLK numeric,
	TOV numeric,
	EFF numeric,
    PRIMARY KEY (index)
);
